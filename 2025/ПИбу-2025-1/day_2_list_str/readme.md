```py
lst = [
    "топот", 
    "топор", 
    "Поп", 
    "Аргентина манит негра",
    "А роза упала на лапу Азора!",
    "Да, искать такси - ад."
]

for s in lst:
    tmp = s.lower()
    t = ''
    for smb in tmp:
        if smb != ' ':
            t += smb
    print(s, t, t == t[::-1])
```
